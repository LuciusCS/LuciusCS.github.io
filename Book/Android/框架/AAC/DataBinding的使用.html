Databinding点击事件的几种实现方式

代码片段

    //BaseActivity中
    public void click(View view){
    }

    public void click1(){
    }

    public void click2(int id){
    }

    //xml文件中
    <variable
        name="activity"
        type=".BaseActivity" />

    <variable
        name="user"
        type=".User" />

1、@{activity.click}

click函数带有参数

    <TextView
         android:layout_width="match_parent"
         android:layout_height="wrap_content"
         android:text="点击事件"
         android:onClick="@{activity.click}"
    />

2、@{()->activity.click()}

click函数无参数

    <TextView
        android:layout_width="match_parent"
        android:layout_height="wrap_content"
        android:text="点击事件"
        android:onClick="@{()->activity.click1()}"
    />

3、@{activity::click}

click带参数

    <TextView
        android:layout_width="match_parent"
        android:layout_height="wrap_content"
        android:text="点击事件"
        android:onClick="@{activity::click}"
    />

4、@{()->activity.click(user.id)}

    <TextView
        android:layout_width="match_parent"
        android:layout_height="wrap_content"
        android:text="点击事件"
        android:onClick="@{()->activity.click2(user.id)}"
    />

5、@{()->activity.click(3)}

    <TextView
        android:layout_width="match_parent"
        android:layout_height="wrap_content"
        android:text="点击事件"
        android:onClick="@{()->activity.click2(3)}"
    />
